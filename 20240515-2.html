#include <stdio.h >
/* 상수 포인터? */
int main() {
	int a;
	int b;
	const int* pa = &a;

	*pa = 3;  // 올바르지 않은 문장
	pa = &b;  // 올바른 문장 
	return 0;
}


/*pa는 상수 포인터이므로 *pa를 통해 가리키는 변수의 값을 변경할 수 없습니다.
따라서 *pa = 3;은 컴파일 오류를 발생시킵니다.
상수 포인터가 가리키는 변수의 값을 변경하는 것은 허용되지 않습니다.

pa는 상수 포인터이지만 pa 자체는 다른 변수의 주소를 가리키도록 변경할 수 있습니다.
따라서 pa = &b; 는 올바른 문장입니다.pa는 이제 b의 주소를 가리킵니다.

절대로 바뀌지 않을 것 같은 값에는 무조건 const 키워드를 붙여주는 습관*/

/* 상수 포인터? */
#include <stdio.h >
int main() {
	int a;
	int b;
	int* const pa = &a;

	*pa = 3;  // 올바르지 않은 문장
	pa = &b;  // 올바른 문장 
	return 0;
}


/*int* const pa = &a; pa의 주소는 못바꾸지만 주소안에 값(변수)는 변경가능*/

int arr[3][3]; 배열은 내부적으로 연속적인 메모리 공간에 할당됩니다. 
이 배열은 3행 3열의 2차원 배열로, 각 요소는 int 형을 가집니다.

이 배열은 메모리 상에서 연속적인 구조로 처리됩니다.
 배열 arr[3][3]은 실제로 1차원 배열로 메모리에 연속적으로 할당됩니다. 예를 들어, 
arr[0][0], arr[0][1], arr[0][2], arr[1][0], arr[1][1], arr[1][2], arr[2][0], arr[2][1], arr[2][2] 순서대로 메모리에 배치됩니다.

따라서, arr[0][0]부터 arr[2][2]까지의 모든 요소는 메모리에서 연속된 위치에 저장되며, 각 요소는 int 변수의 크기에 따라 메모리가 할당됩니다.

문제 2:
int* arr[3]; 배열은 포인터 배열을 나타냅니다. 각 요소는 int 형 포인터입니다.

int* arr[3];는 크기가 3인 배열로, 각 요소는 int 형 변수를 가리키는 포인터입니다. 
이 배열은 각각 int 형 변수를 가리키는 포인터를 세 개 저장할 수 있는 공간을 가지고 있습니다.

이 배열은 다른 주소를 가리키는 포인터를 요소로 가질 수 있으므로, 각 요소가 가리키는 대상은 동적으로 할당될 수 있습니다. 
예를 들어, arr[0], arr[1], arr[2]는 각각 int 형 변수를 가리키는 포인터일 수 있고, 
이들을 이용하여 동적으로 할당된 메모리나 다른 배열의 요소를 가리킬 수 있습니다.

따라서, int* arr[3]; 배열은 각 요소가 int 형 변수를 가리키는 포인터인 포인터 배열을 나타냅니다.
