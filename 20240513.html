#include <stdio.h >

          int main() {
              float f;//12.3456
              int i;
          
              printf("실수를 입력하시오: ");
              if (scanf_s("%f", &f) != 1) {// 12.3456
                  printf("잘못된 입력입니다.\n");
              }
              else {
                  // 입력된 값이 정수인지 확인
                  if ((int)f == f) {
                      // 입력된 값이 정수일 경우
                      i = (int)f; // 실수를 정수로 변환하여 i에 저장
                  }
                  else {
                      // 입력된 값이 실수일 경우
                      i = (int)(f * 100) % 100; // 실수의 소수점 이하 두 자리를 추출하여 i에 저장
                  }// 실수를 100을 곱하여 소수점 이하 두 자리를 정수로 만든 후, 100으로 나누어 추출 
                  //1. 12.3456*100=1234.56 2. (int)정수로 변환 1234 3. 1234를 100으로 나눈 나머지를구함 23 = i
          
                  printf("i = %d\n", i);//23
              }
          
              return 0;
          }
