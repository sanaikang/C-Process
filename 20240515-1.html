/* 포인터도 변수이다 */
#include <stdio.h >
int main() {
	int a;
	int b;
	int* p;

	p = &a;
	*p = 2;
	p = &b;
	*p = 4;

	printf("a : %d \n", a);
	printf("b : %d \n", b);
	return 0;
}
/*int a;, int b;: 정수형 변수 a와 b를 선언합니다.
int* p;: p라는 포인터 변수를 선언합니다. 이 포인터는 정수형 변수의 주소를 저장할 수 있습니다.
p = &a;: 포인터 p에 변수 a의 주소를 할당합니다. &a는 변수 a의 주소를 나타냅니다.
*p = 2;: 포인터 p가 가리키는 주소에 2를 저장합니다. 여기서 *p는 p가 가리키는 주소의 값을 나타냅니다. 따라서 a의 값이 2로 변경됩니다.
p = &b;: 이제 포인터 p에는 변수 b의 주소를 할당합니다.
*p = 4;: 포인터 p가 가리키는 주소에 4를 저장합니다. 따라서 b의 값이 4로 변경됩니다.*/
