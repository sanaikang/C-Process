        #include <stdio.h >

            int main() {
                // 산술 연산
                int a, b;
                a = 10;
                b = 3;
                printf("a + b 는 : %d \n", a + b); // 덧셈 연산 결과 13
                printf("a - b 는 : %d \n", a - b); // 뺄셈 연산 결과 7
                printf("a * b 는 : %d \n", a * b); // 곱셈 연산 결과 30 
                printf("a / b 는 : %d \n", a / b); // 나눗셈 연산 결과 3 (정수 나눗셈이므로 소수부분 버림)
                printf("a %% b 는 : %d \n\n", a % b); // 나머지 연산 결과 1
            
                // 산술 변환
                int c;
                double d;
                c = 10;
                d = 3;
                printf("a / b 는 : %f \n", c / d); // 정수형과 실수형의 연산 결과 3.333333
                printf("b / a 는 : %f \n\n", d / c); // 실수형과 정수형의 연산 결과 0.300000
                // 산술 변환 과정에서 실수형이 정수형보다 큰 자료형이므로 실수 결과가 나옴
            
                // 대입 연산자
                int e = 3;
                e = e + 3;
                printf("e 의 값은 : %d \n\n", e); // 대입 연산자를 통해 e 값이 6이 됨
            
                // 복합 대입 연산자
                int f = 1, g = 1, h = 1, i = 1;
                f = f + 1;     printf("f : %d \n", f); // f에 1을 더한 값 2
                g += 1;        printf("g : %d \n", g); // 복합 대입 연산자 g += 1 을 통해 g 값이 2
                ++h;          printf("h : %d \n", h); // 전위 증감 연산자 ++h를 통해 h 값이 2
                i++;          printf("i : %d \n\n", i); // 후위 증감 연산자 i++를 통해 i 값이 2
            
                // 전위/후위 증감 연산자
                int j = 1;
                printf("++j : %d \n", ++j); // 전위 증감 연산자 ++j를 통해 j 값이 2
                j = 1;
                printf("j++ : %d \n", j++); // 후위 증감 연산자 j++를 통해 먼저 j 값 1 출력 후 1 증가
                printf("j : %d \n\n", j); // j 값은 2
            
                // 비트 연산
                int k = 0xAF, l = 0xB5;
                printf("%x \n", k & l); // AND 연산 결과 a5
                printf("%x \n", k | l); // OR 연산 결과 bf 
                printf("%x \n", k ^ l); // XOR 연산 결과 1a
                printf("%x \n", ~k);    // 비트 반전 연산 결과 ffffff50
                printf("%x \n", k << 2); // 왼쪽 시프트 연산 결과 2bc
                printf("%x \n", l >> 3); // 오른쪽 시프트 연산 결과 16
            
                return 0;
            }
            
            /*뭘 배웠지 ?
            + , -, / , *, =, % 가 무슨 역할을 하는 연산자 인지 배웠습니다.
            &, | , << , >> 가 무슨 역할을 하는 연산자 인지 배웠습니다.
            연산자 우선 순위에 대해 다루었습니다.우선 순위가 헷갈리거나, 복잡한 수식이면 괄호를 적극 활용합니다.*/
